    /**
     * Check if two suggestions are similar
     * @param suggestion1 First suggestion
     * @param suggestion2 Second suggestion
     * @returns Boolean indicating if suggestions are similar
     */
    private isSimilarSuggestion(suggestion1: string, suggestion2: string): boolean {
        // If either is empty, not similar
        if (!suggestion1 || !suggestion2) {
            return false;
        }
        
        // Convert to lowercase and split into words
        const words1 = suggestion1.toLowerCase().split(/\s+/).filter(word => word.length > 3);
        const words2 = suggestion2.toLowerCase().split(/\s+/).filter(word => word.length > 3);
        
        // If either has no significant words, not similar
        if (words1.length === 0 || words2.length === 0) {
            return false;
        }
        
        // Count common words
        const commonWords = words1.filter(word => words2.includes(word));
        
        // Calculate similarity score
        const similarityScore = commonWords.length / Math.min(words1.length, words2.length);
        
        // Consider similar if score is above threshold
        return similarityScore >= 0.3;
    }
